#include <ESP8266WiFi.h>  
#include <WiFiClient.h>  
#include <ThingSpeak.h>  
const char* ssid = "G3_3155";  
const char* password = "perihan1995";  
WiFiClient client;  
unsigned long myChannelNumber = 268913;  
const char * myWriteAPIKey = "RCRU0FDPLXOWRF2S";

//Parmak Nabýz Sensörü Tanýmlamalarý
#define pinKY039 A0
uint8_t pulse;
double alpha = 0.75;
int period = 20;
//double change = 0.0;
//double minval = 0.0;

//Sýcaklýk-Nem Sensörü Tanýmlamalarý
#include <DHT11.h>
#include <DHT.h>
#define DHTPIN D5  
#define DHTTYPE DHT11 
DHT dht(DHTPIN, DHTTYPE);  
uint8_t temperature, humidity;

void setup() {
  dht.begin();
  Serial.begin (115200);
  delay(10);
  // Connect to WiFi network  
  Serial.println();  
  Serial.println();  
  Serial.print("Connecting to ");  
  Serial.println(ssid);  
  WiFi.begin(ssid, password);  
  while (WiFi.status() != WL_CONNECTED)  
  {  
   delay(500);  
   Serial.print(".");  
  }  
  Serial.println("");  
  Serial.println("WiFi connected");  
  // Print the IP address  
  Serial.println(WiFi.localIP());  
  ThingSpeak.begin(client);
}

void loop() {
  
   static double oldValue = 0;
   //static double oldChange = 0;
   int rawValue = analogRead (pinKY039);
   pulse = alpha * oldValue + (1 - alpha) * rawValue;
   oldValue = pulse;
   Serial.print (rawValue);
   Serial.print (",");
   Serial.println (pulse);
   
   temperature = dht.readTemperature();  
   humidity = dht.readHumidity();
   Serial.print("Temperature Value is :");  
   Serial.print(temperature);  
   Serial.println("C");  
   Serial.print("Humidity Value is :");  
   Serial.print(humidity);  
   Serial.println("%");  

   ThingSpeak.setField(1,temperature);
   ThingSpeak.setField(2,humidity);
   ThingSpeak.setField(4,pulse);
   ThingSpeak.writeFields(myChannelNumber, myWriteAPIKey);
  delay(3000);
}
